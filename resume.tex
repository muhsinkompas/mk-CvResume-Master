    \vspace{0.8em}%

%------------------------------------------------------------------------------
% Header with Personal Information (without photo)
%------------------------------------------------------------------------------
\newcommand{\makeheader}[8]{%
    % #1: Name, #2: Title, #3: Address, #4: Phone
    % #5: Email, #6: Website, #7: GitHub, #8: LinkedIn
    \begin{center}
        {\Huge\bfseries\color{primary}\sffamily #1}\\[4pt]
        {\large\color{secondary}#2}\\[8pt]
        \color{textcolor}
        #3 \quad $\bullet$ \quad #4 \quad $\bullet$ \quad \href{mailto:#5}{\color{accent}#5}\\[3pt]
        \href{https://#6}{\color{accent}#6} \quad $\bullet$ \quad 
        \href{https://github.com/#7}{\color{accent}github.com/#7} \quad $\bullet$ \quad 
        \href{https://linkedin.com/in/#8}{\color{accent}linkedin.com/in/#8}
    \end{center}
    \vspace{6pt}
}

%------------------------------------------------------------------------------
% Header with Personal Information (with photo)
%------------------------------------------------------------------------------
\newcommand{\makeheaderwithphoto}[9]{%
    % #1: Name, #2: Title, #3: Address, #4: Phone
    % #5: Email, #6: Website, #7: GitHub, #8: LinkedIn, #9: Photo path
    
    \noindent
    % --- MIDDLE COLUMN (INFO) ---
    \begin{minipage}[c]{0.85\textwidth}
        \centering
        
        % NAME - Two color option:
        {\fontsize{32pt}{1em}\bfseries\color{gray}Your} {\fontsize{32pt}{1em}\bfseries\color{darkgray}Name}\\[0.5em]
        % Single color option (uncomment below and comment above):
        % {\fontsize{32pt}{1em}\bfseries\color{darkgray} #1} \\[0.5em]
        
        % Title
        {\fontsize{13.0pt}{1em}\scshape\color{secondary} #2} \\[0.2em]
        
        % Address
        {\fontsize{9pt}{1em}\itshape\color{textcolor} #3} \\[0.1em]

        % Contact Information with icons
        \fontsize{10pt}{1em}{\includesvg[height=1.0em]{iphone}\space#4 \space \textbar \space \href{mailto:#5}{\includesvg[height=0.8em]{envelope}\space\color{accent}#5} \space \textbar \space \href{https://linkedin.com/in/#8}{\includesvg[height=0.8em]{linkedin}\space\color{accent}linkedin.com/in/#8} \\[0.1em]
        \href{https://#6}{\includesvg[height=0.9em]{website}\space\color{accent}#6} \space \textbar \space \href{https://github.com/#7}{\includesvg[height=0.9em]{github}\space\color{accent}github.com/#7}}
        
    \end{minipage}%
    % --- RIGHT COLUMN (PHOTO) ---
    \begin{minipage}[c]{0.15\textwidth}
        \raggedleft
        \begin{tikzpicture}[baseline=(current bounding box.center)]
            \clip (0,0) circle (1.4cm);
            \node at (0,0) {\includegraphics[width=2.8cm]{#9}}; 
        \end{tikzpicture}
    \end{minipage}
    \vspace{2pt}
}

% Thesis line with link
% Usage: \thesis{Title}{Link}
\newcommand{\thesis}[2]{%
    \vspace{2pt}
    \small\textbf{Thesis: }{\small #1} \textbf{[\href{#2}{\color{accent}Link}]}\\
}

% Thesis line without link
% Usage: \thesisnolink{Title}
\newcommand{\thesisnolink}[1]{%
    \vspace{2pt}
    \small\textbf{Thesis: }{\small #1}\\
}

% Description text
\newcommand{\entrydescription}[1]{%

}
